import React, { useState, useCallback } from "react"

const InputField = ({ label, type, placeholder }) => (
  <div className="mb-4">
    <label className="block mb-2 text-sm font-medium text-white">{label}</label>
    <input
      type={type}
      placeholder={placeholder}
      className="w-full px-3 py-2 text-gray-700 bg-white border rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
    />
  </div>
)

export default function Carousel() {
  const [isFormOpen, setIsFormOpen] = useState(false)

  const toggleForm = useCallback(() => {
    console.log("toggleForm called, current isFormOpen:", isFormOpen)
    setIsFormOpen((prevState) => {
      console.log("Setting isFormOpen to:", !prevState)
      return !prevState
    })
  }, [isFormOpen])

  console.log("Rendering Carousel. isFormOpen:", isFormOpen)

  return (
    <div className="relative w-full max-w-screen-xl mx-auto px-4 pt-16 md:pt-20 bg-white dark:bg-gray-900">
      <div className="relative h-56 sm:h-64 md:h-80 lg:h-96 overflow-hidden rounded-lg shadow-xl">
        <div className="absolute inset-0 flex flex-col items-center justify-center text-center text-white bg-black bg-opacity-50 p-4">
          <h2 className="text-xl sm:text-2xl md:text-3xl lg:text-4xl font-bold mb-4 transition-all duration-300">
            Are you a senior marketing executive leading Global Sports & Events
            partnerships?
          </h2>
          <button
            onClick={() => {
              console.log("Get a quote button clicked")
              toggleForm()
            }}
            className="bg-blue-600 hover:bg-blue-700 text-white font-bold py-2 px-4 rounded-lg transition duration-300 ease-in-out transform hover:scale-105 focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-opacity-50"
          >
            Get a quote
          </button>
        </div>
      </div>

      {/* Sliding Form */}
      <div
        className={`fixed top-0 right-0 w-full sm:w-96 h-full bg-blue-900 shadow-lg transform transition-transform duration-300 ease-in-out z-50 ${
          isFormOpen ? "translate-x-0" : "translate-x-full"
        }`}
      >
        <div className="relative h-full p-8">
          <button
            onClick={toggleForm}
            className="absolute top-4 right-4 text-white hover:text-gray-300"
          >
            <svg
              className="w-6 h-6"
              fill="none"
              stroke="currentColor"
              viewBox="0 0 24 24"
              xmlns="http://www.w3.org/2000/svg"
            >
              <path
                strokeLinecap="round"
                strokeLinejoin="round"
                strokeWidth={2}
                d="M6 18L18 6M6 6l12 12"
              />
            </svg>
          </button>
          <h2 className="text-2xl font-bold text-white mb-6">Get in Touch</h2>
          <form onSubmit={(e) => e.preventDefault()}>
            <InputField
              label="Full Name"
              type="text"
              placeholder="John Doe"
            />
            <InputField
              label="Cell Phone"
              type="tel"
              placeholder="(123) 456-7890"
            />
            <InputField
              label="Email Address"
              type="email"
              placeholder="john@example.com"
            />
            <div className="mb-4">
              <label className="block mb-2 text-sm font-medium text-white">
                Message
              </label>
              <textarea
                className="w-full px-3 py-2 text-gray-700 bg-white border rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-500"
                rows="4"
                placeholder="Your message here..."
              ></textarea>
            </div>
            <button
              type="submit"
              className="w-full bg-blue-600 text-white py-3 px-4 rounded-lg hover:bg-blue-700 transition duration-300 ease-in-out focus:outline-none focus:ring-2 focus:ring-blue-500 focus:ring-opacity-50"
            >
              Submit
            </button>
          </form>
        </div>
      </div>
    </div>
  )
}
